@echo off
setlocal

REM Define variables
set "url=https://www.python.org/ftp/python/2.7/python-2.7.msi"
set "msiFile=python-2.7.msi"
set "installCommand=msiexec /i %msiFile% /quiet"

REM Download the MSI file
echo Downloading %msiFile%...
certutil -urlcache -split -f "%url%" "%msiFile%" >nul 2>&1

REM Install Python silently
echo Installing Python...
%installCommand%

REM Cleanup
del "%msiFile%"

endlocal
